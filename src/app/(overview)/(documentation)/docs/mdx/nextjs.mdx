## Add the library

1. Run the following command:

<InstallLibrary />

2. Add the component to the `app.tsx` file:

> ðŸ’¡ JsonViewer needs `use client` directive.

```tsx {1, 3,8}
"use client";

import { JsonViewer } from "@franmella/json-viewer";

export default function App() {
  return (
    <div>
      <JsonViewer />
    </div>
  );
}
```

3. Pass the JSON data to the component:

```tsx {6-16, 20}
"use client";

import { JsonViewer } from "@franmella/json-viewer";

export default function App() {
  const json = {
    name: "John Doe",
    age: 30,
    email: "test@test.com",
    address: {
      street: "123 Main St",
      city: "Springfield",
      state: "IL",
      zip: "62701",
    },
  };

  return (
    <div>
      <JsonViewer data={json} />
    </div>
  );
}
```
